第三步: 将传入的data数据添加set get属性为后面双向绑定做准备

./src/build.js
class Vue {
    constructor(option) {
      this.el = document.querySelector(option.el)
      this.$data = option.data
      this.Observe(this.$data)
    }
    // 数据监听器也是发布者
    Observe(data) {
        for (let key in data) {
            let value = data[key]
            // configurable 控制是否可以删除 writable 控制是否可以修改(赋值)
            Object.defineProperty(this.$data, key, {
              enumerable: true, // 可枚举
              configurable: true,
              writable: true,
                get: function() {
                    return value
                },
                set: function(newVal) {
                    if (value !== newVal) {
                        value = newVal
                    }
                }
            })
        }
    }
}

完成代码:


class Vue {
    constructor(option) {
        this.el = document.querySelector(option.el)
        this.$data = option.data
        this.dir = {} // 起收集订阅者作用 dir: { test: [], home: []  } 对data的属性收集订阅
        this.Observe(this.$data)
        this.Combile(this.el)
    }
    // 数据监听器也是发布者
    Observe(data) {
        for (let key in data) {
            this.dir[key] = []
            let value = data[key]
            const watcher = this.dir[key]
            // configurable 控制是否可以删除 writable 控制是否可以修改(赋值)
            Object.defineProperty(this.$data, key, {
              enumerable: true, // 可枚举
              configurable: true,
              writable: true,
                get: function() {
                    return value
                },
                set: function(newVal) {
                    if (value !== newVal) {
                        value = newVal
                        watcher.forEach(item => {
                            item.update()
                        })
                    }
                }
            })
        }
    }
    // 编译器，负责指令解析，创建订阅者等
    Combile(el) {
        for (var i = 0; i < el.children.length; i++) {
            const children = el.children[i]
            if (children.children.length) {
                this.Combile(children)
            } else {
                if (children.hasAttribute('v-text')) {
                    var val = children.getAttribute('v-text')
                    this.dir[val].push(new Watcher(children, 'innerHTML', val, this.$data))
                }
                if (children.hasAttribute('v-model')) {
                    const val = children.getAttribute('v-model')
                    this.dir[val].push(new Watcher(children, 'value', val, this.$data))
                    children.addEventListener('input', function(value) {
                        this.$data[val] = children.value
                    }.bind(this)) // 改变this指向 或者使用箭头函数
                }
            }
        }
    }
    // 订阅者，收到数据更新去更新视图
class Watcher {
    constructor(node, type, key, data) {
        this.node = node // 当前节点
        this.type = type // 节点内容
        this.key = key // data内的key值
        this.$data = data // 构造vue时传入的data值
        this.update() // 更新节点的值
    }
    update() {
        this.node[this.type] = this.$data[this.key]
    }
}
